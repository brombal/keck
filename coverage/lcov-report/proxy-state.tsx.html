export interface ObservableFactory {
  makeObservable: (context: ObservableContext) => any;
  handleChange(value: any, identifier: unknown, newValue: any): void;
  createClone(value: any): any;
}
 
interface ObservableContext {
  value: any;
  identifier: any;
  parentContext: ObservableContext | null;
  factory: ObservableFactory;
  observable: any;
  createObservable: (
    value: any,
    identifier: unknown,
    isIntermediate: boolean
  ) => any;
  observeChange: (identifier: unknown) => void;
  subscriptions: Set<any>;
  childContexts: Map<any, ObservableContext>;
}
 
export function createStore<T extends object>(data: T) {
  let isObserving = true;
  let activeCallback: Callback | undefined;
  let isWatchingIntermediates = false;
 
  function createObservableContext(
    thisValue: any,
    parentContext: ObservableContext | null, // Root level will have null parent context
    identifier: any
  ): ObservableContext | undefined {
    if (!thisValue || !observableMap.has(thisValue.constructor)) return;
 
    const ctx: ObservableContext = {
      value: thisValue,
      identifier,
      parentContext,
      subscriptions: new Set(),
      childContexts: new Map(),
      factory: observableMap.get(thisValue.constructor)!,
      observable: null!, // This value is set after the utils are created.
      createObservable(child, identifier, isIntermediate) {
        // If the property is something we know how to observe, return the observable value
        // Get existing child context
        let childContext = ctx.childContexts.get(identifier);
        // If it doesn't exist, try to create it
        if (!childContext) {
          childContext = createObservableContext(child, ctx, identifier);
          if (childContext) ctx.childContexts.set(identifier, childContext);
        }
        if (childContext) {
          // If it's an intermediate, we need to watch it for changes
          if (isIntermediate && isWatchingIntermediates)
            ctx.subscriptions.add(identifier);
          return childContext.observable;
        }
 
        // If it still doesn't exist, it's not observable. Continue...
 
        // If the value is a function, bind it to its parent
        if (typeof child === "function") {
          return child.bind(ctx.observable);
        }
 
        // If it's a non-observable (i.e. a primitive or unknown object type), just subscribe and return
        if (isObserving && activeCallback) ctx.subscriptions.add(identifier);
        return child;
      },
      observeChange(childIdentifier) {
        // Invoke callback if listening
        if (activeCallback && ctx.subscriptions.has(childIdentifier))
          activeCallback();
 
        if (ctx === rootCtx) return;
 
        ctx.value = ctx.factory.createClone(ctx.value);
        ctx.observable = ctx.factory.makeObservable(ctx);
        ctx.parentContext!.factory.handleChange(
          ctx.parentContext!.value,
          identifier,
          ctx.value
        );
        ctx.parentContext!.observeChange(identifier);
      },
    };
    ctx.observable = ctx.factory.makeObservable(ctx);
    return ctx;
  }
 
  const rootCtx = createObservableContext(data, null, null);
 
  return {
    store: rootCtx!.observable as T,
    listen(cb: Callback) {
      isObserving = true;
      activeCallback = cb;
      return () => {
        isObserving = false;
      };
    },
    reset() {
      activeCallback = undefined;
    },
    watchIntermediates(selector: () => void) {
      isWatchingIntermediates = true;
      selector();
      isWatchingIntermediates = false;
    },
  };
}
