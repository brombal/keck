/**
 * The set of supported observable factories. Implement an `ObservableFactory` and add it to this map to add support
 * for custom classes.
 */
export const observableFactories = new Map<
  new (...args: any[]) => any,
  ObservableFactory<any, any>
>();
 
export const getContext = Symbol("getContext");
 
export interface Observable {
  [getContext](): PublicObservableContext;
}
 
/**
 * This interface is used to create observable objects. To create an observable for a class, implement this interface
 * and add it to `observableFactories` using the class as the key.
 */
export interface ObservableFactory<TValue = unknown, TIdentifier = unknown> {
  /**
   * Returns an object that stands in place of the original value, and can be observed.
   */
  makeObservable: (context: ObservableContext<TValue>) => Observable;
 
  /**
   * Applies a change to `value` for the given `identifier`.
   * An example of a change for a plain object would be `value[identifier] = newValue`.
   */
  handleChange(value: TValue, identifier: TIdentifier, newValue: unknown): void;
 
  /**
   * Returns a clone of `value`.
   */
  createClone(value: TValue): unknown;
}
 
type Identifier = unknown;
 
export interface PublicObservableContext<TValue = any, TMeta = any> {
  readonly value: TValue;
  observeIdentifier<T = unknown>(identifier: Identifier, childValue?: T): T;
  modifyIdentifier(identifier: Identifier): void;
}
 
interface ObservableContext<TValue = any, TMeta = any>
  extends PublicObservableContext<TValue, TMeta> {

  identifier: Identifier;
  parentContext: ObservableContext | null;
  childContexts: Map<Identifier, ObservableContext>;
  factory: ObservableFactory;
  _observable: undefined | (unknown & Observable);
  readonly observable: unknown;
}

let i = 0;
 
export function createObservable<T extends object>(
  data: T,
  callback: Callback,
  options?: ObservableOptions
) {
  let storeMeta = allStoresMeta.get(data)!;
  if (!storeMeta) {
    storeMeta = { callbacks: new Set(), values: {} };
    allStoresMeta.set(data, storeMeta);
  }
 
  const observer: Observer = {
    intermediates: false,
    observe: true,
    callback,
    identifiers: {},
  };
 
  function getObservableContext(
    thisValue: unknown,
    parentContext: ObservableContext | null, // Root level will have null parent context
    identifier: Identifier
  ): ObservableContext | undefined {
    if (!thisValue || !observableFactories.has(thisValue.constructor as any)) return undefined;
 
    if (parentContext?.childContexts.has(identifier)) {
      const ctx = parentContext.childContexts.get(identifier)!;
      if (ctx.sharedRef.value !== thisValue) ctx.sharedRef.value = thisValue;
      return ctx;
    }
 
    const parentSymbol = parentContext?.sharedRef.symbol || rootSymbol;
    let sharedRef = storeMeta.values[parentSymbol]?.get(identifier);
    if (!sharedRef) {
      sharedRef = { symbol: Symbol(`ref-${i++}`), value: thisValue, observables: new Set() };
      storeMeta.values[parentSymbol] ||= new Map();
      storeMeta.values[parentSymbol].set(identifier, sharedRef);
    }
 
    const ctx: ObservableContext = {
      sharedRef,
      get value() {
        return this.sharedRef.value;
      },
      identifier,
      parentContext,
      childContexts: new Map(),
      factory: observableFactories.get(thisValue.constructor as any)!,
      _observable: undefined,
      get observable() {
        if (!this._observable || !this.sharedRef.observables.has(this._observable))
          this._observable = this.factory.makeObservable(this);
        // this._observable[__raw] = this.sharedRef.value;
        this.sharedRef.observables.add(this._observable);
        return this._observable;
      },
 
      observeIdentifier(identifier, childValue) {
        // If the value is a function, bind it to its parent
        if (typeof childValue === "function") {
          return childValue.bind(ctx.observable);
        }
 
        function createObserver() {
          if (!observer.observe) return;
          let observedIds = observer.identifiers[ctx.sharedRef.symbol] || new Set();
          observedIds.add(identifier);
          observer.identifiers[ctx.sharedRef.symbol] = observedIds;
        }
 
        if (childValue) {
          // If the property is something we know how to observe, return the observable value
          const childContext = getObservableContext(childValue, ctx, identifier);
          if (childContext) {
            if (observer.intermediates) createObserver();
            return childContext.observable;
          }
        }
 
        // If it's a non-observable (i.e. a primitive or unknown object type), just observe and return
        createObserver();
        return childValue;
      },
 
      modifyIdentifier(childIdentifier) {
        if (ctx !== rootCtx) {
          ctx.sharedRef.value = ctx.factory.createClone(ctx.sharedRef.value);
          ctx.sharedRef.observables.clear(); //.delete(ctx._observable);
          ctx._observable = undefined;
        }
 
        storeMeta.callbacks.forEach(({ callback, identifiers }) => {
          if (callback && identifiers[ctx.sharedRef.symbol]?.has(childIdentifier)) {
            callback(ctx.sharedRef.value, childIdentifier);
          }
        });
 
        ctx.parentContext?.factory.handleChange(
          ctx.parentContext!.sharedRef.value,
          ctx.identifier,
          ctx.sharedRef.value
        );
 
        ctx.parentContext?.modifyIdentifier(ctx.identifier);
      },
    };
    parentContext?.childContexts.set(identifier, ctx);
    return ctx;
  }
 
  const rootCtx = getObservableContext(data, null, null)!;
 
  function observe(options?: ObservableOptions) {
    if (!observer.observe) observer.identifiers = {};
    observer.observe = true;
    observer.intermediates = !!options?.intermediates;
    storeMeta.callbacks.add(observer);
  }
  observe(options);
 
  return {
    store: rootCtx!.observable as T,
    observe,
    unobserve() {
      observer.observe = false;
    },
    disable() {
      observer.observe = false;
      observer.callback = undefined;
    },
    enable() {
      observer.callback = callback;
    },
  };
}
 
export function unwrap<T>(observable: T): T {
  const ctx = (observable as Observable)?.[getContext]?.() as ObservableContext;
  ctx?.parentContext?.observeIdentifier(ctx.identifier, ctx.value);
  return ctx?.value || observable;
}
